#include <iostream>
using namespace std;

int main() {
    // copying a pointer
    int num = 5;
    int a = num;
    a++;
    
    int *p = &num; // iske liye amp dalna hota hai
    (*p)++;
    int *q = p; // for copy pointer koi bhi star ya amp nhi dalna
    cout<<p<<"   "<<q;
	
}


POINTER ARITHMETIC
THE FIRST CASE INCREASES THE VALUE OF THE VARIABLE WHOSE LOCATION IS STORED IN THE POINTER
#include <iostream>
using namespace std;

int main() {
    // pointer arithmetic
    int i = 3;
    int *ptr = &i;
    cout<<(*ptr)++<<endl;;
    cout<<*ptr;
    
	
}
OUTPUT ->
3
4
THIS SECOND IMPLEMENTATION SHOWS THAT THE LOCATION OF THE POINTER IS ONLY INCREMENTING 
#include <iostream>
using namespace std;

int main() {
    // pointer arithmetic
    int i = 3;
    int *ptr = &i;

}
